// 1. Write a program that takes a character (number or string)
// in a variable & checks whether the given input is a number,
// uppercase letter, or lowercase letter.

var char = prompt("Enter a character:");
if (!isNaN(char)) {
alert("You entered a number.");
} else if (char >= 'A' && char <= 'Z') {
alert("You entered an uppercase letter.");
} else if (char >= 'a' && char <= 'z') {
alert("You entered a lowercase letter.");
} else {
alert("Invalid input.");
}

// 4. Write a program that takes a character (i.e., string of length 1)
// and returns true if it is a vowel, false otherwise.

} else {
alert("False - It is not a vowel.");
